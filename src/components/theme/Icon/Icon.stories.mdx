# Icon

You can load icons from Volto with:

```jsx static
import { Icon } from '@plone/volto/components';
import addIcon from '@plone/volto/icons/add.svg';

const IconTest = () => <Icon name={addIcon} size="18px" />;
```

export const iconElements = Object.keys(icons).map((iconName) => {
  const icon = icons[iconName];
  return (
    <center
      style={{
        float: 'left',
        width: '150px',
        display: 'inline-block',
        height: '100px',
      }}
    >
      <IconComponent key={iconName} name={icon} /> <br />
      {`${iconName}.svg`}

  );
});

export const Icon = (args) => (
